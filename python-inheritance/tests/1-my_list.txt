# Test case for MyList class

# Importing the MyList class
from 1-my_list import MyList

# Creating an instance of MyList
my_list = MyList()

# Appending elements to the list
my_list.append(1)
my_list.append(4)
my_list.append(2)
my_list.append(3)
my_list.append(5)

# Testing the regular print
>>> print(my_list)
[1, 4, 2, 3, 5]

# Testing the print_sorted method
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]

# Testing that the original list remains unchanged
>>> print(my_list)
[1, 4, 2, 3, 5]
